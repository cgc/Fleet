// NOTE: this must be registered in main with signal(SIGINT, fleet_interrupt_handler);

if(signum == SIGINT) {
CTRL_C = true;
}
else if(signum == SIGHUP) {
// do nothing -- defaultly Fleet mutes SIGHUP so that commands left in the terminal will continue

}
