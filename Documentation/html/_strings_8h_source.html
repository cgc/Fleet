146  // all of these get precomputed at compile time
147  static const float log_add_p = log(add_p);
148  static const float log_del_p = log(del_p);
149  static const float log_1madd_p = log(1.0-add_p);
150  static const float log_1mdel_p = log(1.0-del_p);
151 
152 
153  // Well we can always delete the whole thing and add on the remainder
154  float lp = log_del_p*x.length() + // we don't add log_1mdel_p again here since we can't delete past the beginning
155  (log_add_p-log_alphabet)*y.length() + log_1madd_p;
156

158  // here we index over mi, the length of the string that so far is equal
159  for(size_t mi=1;mi<=std::min(x.length(),y.length());mi++){
160  if(x[mi-1] == y[mi-1]) {
161  lp = logplusexp(lp, log_del_p*(x.length()-mi) + log_1mdel_p +
162  (log_add_p-log_alphabet)*(y.length()-mi) + log_1madd_p);
163  }
164  else {
165  break;
166  }
167  }

331 double p_KashyapOommen1984_edit(const std::string x, const std::string y, const double perr, const size_t nalphabet) {
332  // From Kashyap & Oommen, 1984
333  // perr gets used here to define the probability of each of these operations
334  // BUT Note here that we use the logs of these
335 
336  const int m = x.length(); // these must be ints or some negatives below go bad
337  const int n = y.length();
338  const int T = std::max(m,n); // max insertions allowed
339  const int E = std::max(m,n); // max deletions allowed
340  const int R = std::min(m,n);
341 
342  const double lp_insert = -log(nalphabet); // P(y_t|lambda) - probability of generating a single character from nothing
343  const double lp_delete = log(perr); // P(phi|d_x) -- probability of deleting
344 
345  Vector3D<double> W(T+1,E+1,T+1);
346 
347  // set to 1 in log space
348  W(0,0,0) = 0.0;
349 
350  // indexes into y and x, and returns the swap probability
351  // when they are different. Also corrects so i,j are 0-indexed
352  // instead of 1-indexed, as they are below.
353  // Note that when we are correct, we sum up both ways of getting the answer
354  const auto Sab = [&](int i, int j) -> double {
355  return log(perr / nalphabet + (1.0-perr)*(y[i-1]==x[j-1] ? 1 : 0));
356  };
357 
358  for(int t=1;t<=T;t++)
359  W(t,0,0) = W(t-1,0,0) + lp_insert;
360 
361  for(int d=1;d<=E;d++)
362  W(0,d,0) = W(0,d-1,0) + lp_delete;
363 
364  for(int s=1;s<=R;s++)
365  W(0,0,s) = W(0,0,s-1) + Sab(s-1,s-1);
366 
367  for(int t=1;t<=T;t++)
368  for(int d=1;d<=E;d++)
369  W(t,d,0) = logplusexp(W(t-1,d,0)+lp_insert,
370  W(t,d-1,0)+lp_delete);
371 
372  for(int t=1;t<=T;t++)
373  for(int s=1;s<=n-t;s++)
374  W(t,0,s) = logplusexp(W(t-1,0,s)+lp_insert,
375  W(t,0,s-1)+Sab(s+t-1,s-1));
376 
377  for(int d=1;d<=E;d++)
378  for(int s=1;s<=m-d;s++)
379  W(0,d,s) = logplusexp(W(0,d-1,s)+lp_delete,
380  W(0,d,s-1)+Sab(s-1,s+d-1));
381 
382  for(int t=1;t<=T;t++)
383  for(int d=1;d<=E;d++)
384  for(int s=1;s<=std::min(n-t,m-d);s++)
385  W(t,d,s) = logplusexp(W(t-1,d,s)+lp_insert,
386  logplusexp(W(t,d-1,s)+lp_delete,
387  W(t,d,s-1)+Sab(t+s-1,d+s-1))); // shoot me in the face
388 
389 
390  // now we sum up
391  double lp_yGx = -infinity; // p(Y|X)
392  for(int t=std::max(0,n-m); t<=std::min(T,E+n-m);t++){
393  double qt = lpmf_geometric(t+1,1.0-perr); // probability of t insertions -- but must use t+1 to allow t=0 (geometric is defined on 1,2,3,...)
394  lp_yGx = logplusexp(lp_yGx, W(t,m-n+t,n-t) + qt + lfactorial(m)+lfactorial(t)-lfactorial(m+t));
395  }
396  return lp_yGx;
