331  double search_one(this_t* parent, HYP& current) {
332  // recurse down the tree, building and/or evaluating as needed. This has a nice format that saves space -- Nodes need
333  // not store parents since they can be passed recursively here. Because this returns the score, it can be propagated
334  // back up the tree.
335 
336  mylock.lock();
337 
338  if(DEBUG_MCTS) DEBUG("MCTS SEARCH ONE ", this, "\t["+current.string()+"] ", nvisits);
339 
340  // now if we get here, we must have at least one open child
341  // so we can pick the best and recurse
342 
343  ++nvisits; // increment our visit count on our way down so other threads don't follow
344 
345  auto neigh = current.neighbors();
346 
347  if(neigh == 0) {
348  // if I am a terminal
349 
350  open = false; // make sure nobody else takes this one
351 
352  // if its a terminal, compute the posterior
353  current.compute_posterior(*data);
354  add_sample(current.likelihood);
355  (*callback)(current);
356 
357  mylock.unlock();
358 
359  return current.likelihood;
360  }
361  else {
362

364  // TODO: This is inefficient because it copies current a bunch of times...
365  if(children.size() == 0) {
366  for(size_t k=0;k<neigh;k++) {
367  HYP kc = current; kc.expand_to_neighbor(k);
368  children.emplace_back(kc, this, k);
369  }
