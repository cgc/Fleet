356  hposterior.col(i) = lognormalize(hposterior.col(i)).array().exp();
357  }

361  // is non-ideal

365  for(size_t i=0;i<human_data.size();i++) {

368  // because its very slow

370  for(int h=0;h<hposterior.rows();h++) {

372 
373  for(const auto& mp : P->at(h,i)) {
374  float p = hposterior(h,i) * mp.second;
375  // map 0 for nonexisting doubles

381  auto& di = human_data[i];
382  for(const auto& r : di.responses) {
383  ll += log( (1.0-alpha)*di.chance + alpha*model_predictions[r.first]) * r.second; // log probability times the number of times they answered this

443  Vector hypothesis_prior(Matrix& C) {
444  // take a matrix of counts (each row is a hypothesis)
445  // and return a prior for each hypothesis under my own X
446  // (If this was just a PCFG, which its not, we'd use something like lognormalize(C*proposal.getX()))
447 
448  Vector out(C.rows()); // one for each hypothesis
449 
450  // get the marginal probability of the counts via dirichlet-multinomial
451  Vector allA = logA.value.array().exp(); // translate [-inf,inf] into [0,inf]
452 
453  #pragma omp parallel for
454  for(auto i=0;i<C.rows();i++) {
455 
456  double lp = 0.0;
457  size_t offset = 0; //
458  for(size_t nt=0;nt<grammar->count_nonterminals();nt++) { // each nonterminal in the grammar is a DM
459  size_t nrules = grammar->count_rules( (nonterminal_t) nt);
460  if(nrules > 1) { // don't need to do anything if only one rule (but must incremnet offset)
461  Vector a = eigenslice(allA,offset,nrules); // TODO: seqN doesn't seem to work with this eigen version
462  Vector c = eigenslice(C.row(i),offset,nrules);
463  double n = a.sum(); assert(n > 0.0);
464  double c0 = c.sum();
465  if(c0 != 0.0) { // TODO: Check this...
466  // NOTE: std::lgamma here is not thread safe, but by my read it should be when the argument is positive
467  lp += std::lgamma(n+1) + std::lgamma(c0) - std::lgamma(n+c0)
468  + (c.array() + a.array()).array().lgamma().sum()
469  - (c.array()+1).array().lgamma().sum()
470  - a.array().lgamma().sum();
471  }
472  }
473  offset += nrules;
474  }
475 
476  out(i) = lp;
477  }
478 
479  return out;
