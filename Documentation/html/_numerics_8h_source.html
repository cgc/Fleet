82 T logplusexp(const T a, const T b) {
83  // An approximate logplusexp -- the check on z<-25 makes it much faster since it saves many log,exp operations
84  // It is easy to derive a good polynomial approximation that is a bit faster (using sollya) on [-25,0] but that appears
85  // not to be worth it at this point.
86 
87  if (a == -infinity) return b;
88  else if(b == -infinity) return a;
89 
90  T mx = std::max(a,b);
91 
92  // compute a bound here at such that log1p(exp(z)) doesn't affect the result much (up to 1e-6)
93  // for floats, this ends up being about -13.8
94  // TODO: I Wish we could make this constexpr, but the math functions are not constexpr. Static is fine though
95  // it should only be run once at the start.
96  static const float breakout = get_log1pexp_breakout_bound<T>(1e-6, +[](T x) -> T { return log1p(exp(x)); });
97 
98 
99  T z = std::min(a,b)-mx;
100  if(z < breakout) {
101  return mx; // save us from having to do anything
102  }
103  else {
104  return mx + log1p(exp(z)); // mx + log(exp(a-mx)+exp(b-mx));
105  }
106 }
