254  virtual int sample_child_index(HYP& current) {
255 
256  assert(this->children.size() > 0) ;
257 
258  int neigh = current.neighbors();
259 
260  // probability of expanding to each child
261  std::vector<double> children_lps(neigh, -infinity);
262 
263  // first, load up children_lps with whether they have been visited
264  bool all_visited = true;
265  for(int k=0;k<neigh;k++) {
266  if(this->children[k].open and this->children[k].nvisits == 0) {
267  all_visited = false;
268  children_lps[k] = current.neighbor_prior(k);
269  }
270  }
271 
272  // if all the neighbors have been visited, we'll overwrite everything
273  // with UCT
274  if(all_visited) {
275  for(int k=0;k<neigh;k++) {
276  if(this->children[k].open){
277  children_lps[k] = (this->child(k).nvisits == 0 ? 0.0 : (this->max / this->child(k).max) +
278  FleetArgs::explore * sqrt(log(this->nvisits)/this->children[k].nvisits));
279  }
280  }
281  }
282 
283  // for(int k=0;k<neigh;k++) {
284  // CERR k TAB all_visited TAB children_lps[k] TAB current.neighbor_prior(k) TAB current.string() ENDL;
285  // }
286 
287  // someitmes we'll get all NaNs, which is bad new sfor sampling
288  bool allNaN = true;
289  for(int k=0;k<neigh;k++) {
290  if(this->children[k].open and not std::isnan(children_lps[k])) {
291  allNaN = false;
292  break;
293  }
294  }

388  if(DEBUG_MCTS) DEBUG("PartialMCTSNode SEARCH ONE ", this, "\t["+current.string()+"] ", this->nvisits);
389 
390  auto c = this->descend_to_childless(current); //sets current and returns the node.
391 
392  // if we are a terminal

394  c->process_evaluable(current);
395  }

398  c->add_children(current);

402  }

411  virtual void playout(HYP& current) {
412  if(current.is_evaluable()) {
413  this->process_evaluable(current);
414  }
415  else {
416 
417  // keep track of max since that's what we'll store
418  double mx = -infinity;
