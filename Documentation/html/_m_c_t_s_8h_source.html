264  virtual void descend(HYP& current) {
265  int neigh = current.neighbors();
266  std::vector<double> children_lps(neigh, -infinity);
267  for(int k=0;k<neigh;k++) {
268  if(this->child(k).open){
271  children_lps[k] = current.neighbor_prior(k) +
272  (this->child(k).nvisits == 0 ? lse-1.0 : this->child(k).lse-log(this->child(k).nvisits)) / explore;
273  }
274  }
275 
276  // choose an index into children
277  int idx = sample_int_lp(neigh, [&](const int i) -> double {return children_lps[i];} ).first;
278 
279  // expand
280  current.expand_to_neighbor(idx); // idx here gives which expansion we follow
281 
282  // and recurse down
283  this->child(idx).search_one(current);
284  }

292  //
293 
294  if(DEBUG_MCTS) DEBUG("MCTS SEARCH ONE ", this, "\t["+current.string()+"] ", nvisits);
295 
296  ++nvisits; // increment our visit count on our way down so other threads don't follow

299 
300  // if we can evaluate this

302  process_evaluable(current);

306  if(this->nchildren() != (size_t) neigh) {
