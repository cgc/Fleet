268  virtual void descend(HYP& current) {
269  int neigh = current.neighbors();
270  std::vector<double> children_lps(neigh, -infinity);
271  for(int k=0;k<neigh;k++) {
272  if(this->child(k).open){
275  children_lps[k] = current.neighbor_prior(k) +
276  (this->child(k).nvisits == 0 ? lse-1.0 : this->child(k).lse-log(this->child(k).nvisits)) / this->explore;
277  }
278  }
279 
280  // choose an index into children
281  int idx = sample_int_lp(neigh, [&](const int i) -> double {return children_lps[i];} ).first;
282 
283  // expand
284  current.expand_to_neighbor(idx); // idx here gives which expansion we follow
285 
286  // and recurse down
287  this->child(idx).search_one(current);
288  }

295  void search_one(HYP& current) {
296  //
297 
298  if(DEBUG_MCTS) DEBUG("MCTS SEARCH ONE ", this, "\t["+current.string()+"] ", nvisits);
299 
300  ++nvisits; // increment our visit count on our way down so other threads don't follow
301 
302  int neigh = current.neighbors();
303 
304  // if we can evaluate this
305  if(current.is_evaluable()) {
306  process_evaluable(current);
307  }
308 
309  // if I don't have all of my children, add them
310  if(this->nchildren() != (size_t) neigh) {
311  process_add_children(current);
312  }
313 
314  // otherwise keep going down the tree
315  if(neigh > 0) {
316  descend(current);
317  }
318 
319  } // end search

355  virtual void descend(HYP& current) override {
356 
357  // check if all my children have been visited
358 
359  int neigh = current.neighbors();
360 
361  // if everyone has been visited, we'll descend with a UCT-like rule
362  if(this->all_children_visited()) {
363  std::vector<double> children_lps(neigh, -infinity);
364  for(int k=0;k<neigh;k++) {
365  if(this->children[k].open){
366  children_lps[k] = current.neighbor_prior(k) +
367  (this->children[k].nvisits == 0 ? 0.0 : this->children[k].max + this->explore*sqrt(log(1+this->nvisits)/(1+this->children[k].nvisits)));
368  }
369  }
370 
371  int idx = arg_max_int(neigh, [&](const int i) -> double {return children_lps[i];} ).first;
372  current.expand_to_neighbor(idx); // idx here gives which expansion we follow
373  this->children[idx].search_one(current);
374  }
375  else {
376 
377  // otherwise choose the highest prior one that is unvisited and run playout (which users must define)
378  int idx = arg_max_int(neigh, [&](const int k) -> double { return (this->children[k].nvisits == 0 ? current.neighbor_prior(k) : -infinity); } ).first;
379  current.expand_to_neighbor(idx); // idx here gives which expansion we follow
380 
381  // NOTE: here we do not search_one -- we just stop at this node -- adding one expansion to children
382  // we have to call children[idx] here because otherwise it won't get the sample
383  this->children[idx].nvisits++; // since it's not counted in add_sample and we don't search_one on it
384  this->children[idx].playout(current);
385  }
