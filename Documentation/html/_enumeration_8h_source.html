9 template<typename Grammar_t>
10 Node expand_from_integer(Grammar_t* g, nonterminal_t nt, IntegerizedStack& is) {
11  // this is a handy function to enumerate trees produced by the grammar.
12  // This works by encoding each tree into an integer using some "pairing functions"
13  // (see https://arxiv.org/pdf/1706.04129.pdf)
14  // Specifically, here is how we encode a tree: if it's a terminal,
15  // encode the index of the terminal in the grammar.
16  // If it's a nonterminal, encode the rule with a modular pairing function
17  // and then use Rosenberg-Strong encoding to encode each of the children.
18  // In this way, we can make a 1-1 pairing between trees and integers,
19  // and therefore store trees as integers in a reversible way
20  // as well as enumerate by giving this integers.
21 
22  // NOTE: for now this doesn't work when nt only has finitely many expansions/trees
23  // below it. Otherwise we'll get an assertion error eventually.
24 
25  enumerationidx_t numterm = g->count_terminals(nt);
26  if(is.get_value() < numterm) {
27  return g->makeNode(g->get_rule(nt, is.get_value())); // whatever terminal we wanted
28  }
29  else {
30  // Otherwise it is encoding numterm+which_terminal
31  is -= numterm;
32 
33  auto ri = is.pop(g->count_nonterminals(nt)); // this already includes the shift from all the nonterminals
34 
35  Rule* r = g->get_rule(nt, ri+numterm); // shift index from terminals (because they are first!)
36  Node out = g->makeNode(r);
37  for(size_t i=0;i<r->N;i++) {
38  // note that this recurses on the enumerationidx_t format -- so it makes a new IntegerizedStack for each child
39  out.set_child(i, expand_from_integer(g, r->type(i), i==r->N-1 ? is.get_value() : is.pop()) ) ;
40  }
41 
42  return out;
43  }

55 template<typename Grammar_t>
56 enumerationidx_t compute_enumeration_order(Grammar_t* g, const Node& n) {
57  // inverse of the above function -- what order would we be enumerated in?
58  if(n.nchildren() == 0) {
59  return g->get_index_of(n.rule);
60  }
61  else {
62  // here we work backwards to encode
63  nonterminal_t nt = n.rule->nt;
64  enumerationidx_t numterm = g->count_terminals(nt);
65 
66  IntegerizedStack is(compute_enumeration_order(g, n.child(n.rule->N-1)));
67  for(int i=n.rule->N-2;i>=0;i--) {
68  is.push(compute_enumeration_order(g, n.child(i)));
69  }
70  is.push(g->get_index_of(n.rule)-numterm, g->count_nonterminals(nt));
71  is += numterm;
72  return is.get_value();
73  }
