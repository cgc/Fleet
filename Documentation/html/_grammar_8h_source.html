23  /*
24  * A grammar stores all of the rules associated with any kind of nonterminal and permits us
25  * to sample as well as compute log probabilities.
26  * A grammar also is nwo required to store them in a fixed (sorted) order that is guaranteed
27  * not to change, adn that puts terminals first and (lower priority) high probability first
28  * as determined by Rule's sort order.
29  *
30  * Note that Primitives are used to initialize a grammar, and they get "parsed" by Gramar.add
31  * to store them in the right places according to their return types (the index comes from
32  * the index of each type in GRAMMAR_TYPES).
33  * The trees that Grammar generates use nt<T>() -> size_t to represent types, not the types
34  * themselves.
35  * The trees also use Primitive.op (a size_t) to represent operations
36  *
37  * The template args GRAMMAR_TYPES stores the types used in this grammar. This is inherited
38  * by LOTHypothesis, which also passes them along to VirtualMachineState (in the form of
39  * a Tuple). So the types used and order are fixed/standardized in the grammar
40  */

564  Node expand_from_integer(nonterminal_t nt, IntegerizedStack& is) const {
565  // this is a handy function to enumerate trees produced by the grammar.
566  // This works by encoding each tree into an integer using some "pairing functions"
567  // (see https://arxiv.org/pdf/1706.04129.pdf)
568  // Specifically, here is how we encode a tree: if it's a terminal,
569  // encode the index of the terminal in the grammar.
570  // If it's a nonterminal, encode the rule with a modular pairing function
571  // and then use Rosenberg-Strong encoding to encode each of the children.
572  // In this way, we can make a 1-1 pairing between trees and integers,
573  // and therefore store trees as integers in a reversible way
574  // as well as enumerate by giving this integers.
575 
576  // NOTE: for now this doesn't work when nt only has finitely many expansions/trees
577  // below it. Otherwise we'll get an assertion error eventually.
578 
579  enumerationidx_t numterm = count_terminals(nt);
580  if(is.get_value() < numterm) {
581  return makeNode(this->get_rule(nt, is.get_value())); // whatever terminal we wanted
582  }
583  else {
584 
585  is -= numterm; // remove this from is
586  auto ri = is.pop(count_nonterminals(nt));
587  Rule* r = this->get_rule(nt, ri+numterm); // shift index from terminals (because they are first!)
588  Node out = makeNode(r);
589  for(size_t i=0;i<r->N;i++) {
590  // note that this recurses on the z format -- so it makes a new IntegerizedStack for each child
591  out.set_child(i, expand_from_integer(r->type(i), i==r->N-1 ? is.get_value() : is.pop()) ) ;
592  }
593  return out;
594  }
