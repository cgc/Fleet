880  void expand_to_neighbor(Node& node, int& which) {
881  // here we find the neighbor indicated by which and expand it into the which'th neighbor
882  // to do this, we loop through until which is less than the number of neighbors,
883  // and then it must specify which expansion we want to take. This means that when we
884  // skip a nullptr, we have to subtract from it the number of neighbors (expansions)
885  // we could have taken.
886  for(size_t i=0;i<node.rule->N;i++){
887  if(node.child(i).is_null()) {
888  int c = count_rules(node.rule->type(i));
889  if(which >= 0 and which < c) {
890  auto r = get_rule(node.rule->type(i), (size_t)which);
891  node.set_child(i, makeNode(r));
892  }
893  which -= c;
894  }
895  else { // otherwise we have to process that which
896  expand_to_neighbor(node.child(i), which);
897  }
898  }

991  Node simple_parse(std::string s) {
992  //PRINTN("Parsing ", s);
993 
994  // remove the lambda x. if its there
995  if(s.substr(0,3) == LAMBDAXDOT_STRING) s.erase(0,3);
996  // remove leading whitespace
997  while(s.at(0) == ' ' or s.at(0) == '\t') s.erase(0,1);
998  // remove trailing whitespace
999  while(s.at(s.size()-1) == ' ' or s.at(s.size()-1) == '\t') s.erase(s.size()-1,1);
1000 
1001  // use the above function to find chunks
1002  auto [open, commas, close] = find_open_commas_close(s);
1003 
1004  // if it's a terminal
1005  if(open == -1) {
1006  assert(commas.size()==0 and close==-1);
1007  auto r = this->get_rule(s);
1008  return this->makeNode(r);
1009  }
1010  else if(close == open+1) { // special case of "f()"
1011  return this->makeNode(get_rule(s)); // the whole string is what we want and its a terminal
1012  }
1013  else {
1014  assert(close != -1);
1015 
1016  // recover the rule format -- no spaces, etc.
1017  std::string fmt = s.substr(0,open) + "(%s";
1018  for(auto& c : commas) {
1019  UNUSED(c);
1020  fmt += ",%s";
1021  }
1022  fmt += ")";
1023 
1024  // find the rule for this format
1025  auto r = this->get_rule(fmt);
1026  auto out = this->makeNode(r);
1027 
1028  int prev=open+1;
1029  int ci=0;
1030  for(auto& c : commas) {
1031  auto child_string = s.substr(prev,c-prev);
1032  out.set_child(ci,simple_parse(child_string));
1033  prev = c+1;
1034  ci++;
1035  }
1036 
1037  // and the last child
1038  auto child_string = s.substr(prev,close-prev);
1039  out.set_child(ci,simple_parse(child_string));
1040 
1041  return out;
1042  }
1043  }
